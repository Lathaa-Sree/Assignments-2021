def is_palindrome(string):
    temp = " "
    for i in string:
        temp = i + temp
    if temp == string:
        return True
    return False

def sqrt_of_numbers(num):
    if num >= 0:
        return math.sqrt(num)
    r = str(math.sqrt((abs(num))))+' i'
    return r

def even_sort(arr):
    even = []
    odd = []
    for i in arr:
        if (i % 2 == 0):
            even.append(i)
        else:
            odd.append(i)
    even.sort()
    odd.sort()
    even.extend(odd)
    return even
